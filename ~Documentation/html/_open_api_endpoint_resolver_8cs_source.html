19 public override MgcxmHttpEndpoint ResolveUrl(string url, MgcxmHttpRequest request)
20 {
21 if (OpenApiFramework == null)
22 {
23 Logger.Error("Cannot resolve url because the OpenApiFramework is null.");
24 return null!;
25 }
26
27 if (!url.StartsWith(OpenApiEndpointStructure!.ApiPath))
28 throw new OpenApiEndpointSearchException($"Cannot find '{url}'. The url does not start with '{OpenApiEndpointStructure.ApiPath}' and is not a valid OpenAPI endpoint.");
29
30 foreach (var vEndpoint in OpenApiEndpointStructure.Endpoints)
31 {
32 Logger.Trace("Resolving " + vEndpoint.Key.ToString());
33
34 // resolve the endpoint
35 bool isMatched = MgcxmHttpEndpoint.Matches(url, vEndpoint.Value, request);
36 if (isMatched)
37 {
38 Logger.Trace("Url matching succeeded.");
39 Logger.Trace($"Endpoint = {vEndpoint.Key.ToString()}, Url = {url}");
40 return vEndpoint.Value;
41 }
42 }
43
44 Logger.Error("Cannot resolve url because no endpoints were found.");
45 return null!;
46 }
47
48 public override void ResolveAll()
49 {
50 foreach (var subpartController in _endpointSubpartControllers)
51 {
52 var type = subpartController.GetType();
53 Logger.Trace($"Resolving subpart '{type.FullName}'");
54
55 AttributeResolver.ResolveAllMethod<HttpEndpointAttribute>(
56 type,
57 typeof(HttpEndpointAttribute),
58 (method, attr) =>
59 {
60 if (subpartController.AllEndpoints.Any(x => x.Url == attr.Url))
61 {
62 Logger.Error($"Cannot resolve subpart - there is already a endpoint with the url '{attr.Url}'");
63 return; // cannot add endpoints with same url
64 }
65
66
67 OpenApiEndpointStructure!.AddEndpoint(EndpointUrl.FromString<OpenApiUrl>(attr.Url), attr.HttpMethod,
68 (req, res) =>
69 {
70 var parameters = method.GetParameters();
71 bool isInvalidSig = parameters.Count() != 2 ||
72 (parameters[0].ParameterType != typeof(MgcxmHttpRequest) &&
73 parameters[1].ParameterType != typeof(MgcxmHttpResponse));
74
75 if (isInvalidSig) return; // cannot add endpoints with invalid signatures
76 method.Invoke(subpartController, new object[] { req, res });
77 }, HttpListener);
78 });
79 }
80 }
