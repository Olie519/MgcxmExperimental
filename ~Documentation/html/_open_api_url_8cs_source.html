20 public override string ToString()
21 {
22 string format = string.Format("/api/{0}/v{1}", _controller, _version);
23 if (_subParts.Length > 0)
24 format += string.Format("/{0}", string.Join("/", _subParts));
25

29 public override void FromString(string url)
30 {
31 // /api/{_controller}/v{_version}/{_subParts}
32
33 try
34 {
35 // remove the api path
36 string nUrl = url.Remove(0, 5);
37
38 // get controller
39 string[] parts = nUrl.Split("/");
40 _controller = parts[0];
41
42 // parse version
43 if (!uint.TryParse(parts[1].Replace("v", string.Empty), out uint version))
44 throw new OpenApiUrlFromStringException("Cannot parse version. Given URL = " + url);
45 _version = version;
46
47 // get subparts
48 List<string> subparts = new List<string>();
49 for (int i = 2; i < parts.Length; i++)
50 subparts.Add(parts[i]);
51 _subParts = subparts.ToArray();
52 }
53 catch (Exception exception)
54 {
55 Logger.Error($"Failed to parse OpenApiUrl: ({exception.GetType().GetSafeName()}): {exception}");
56 }
57 }
58
59 public string Controller => _controller;
60 public uint Version => _version;
61 public string[] SubParts => _subParts;
