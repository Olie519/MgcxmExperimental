292 private static string CloseNestedPastelStrings(string input, Color color, ColorPlane colorPlane)
293 {
294 var closedString = CloseNestedPastelStringRegex1().Replace(input, _formatStringEnd);
295
296 closedString = CloseNestedPastelStringRegex2().Replace(closedString, $"{_formatStringEnd}$0");
297 closedString = _closeNestedPastelStringRegex3[colorPlane].Replace(closedString, $"$0{string.Format(_formatStringPartial, _planeFormatModifiers[colorPlane], color.R, color.G, color.B)}");
298
299 return closedString;
300 }
