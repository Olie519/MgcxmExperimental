116 public float GetSeconds()
117 {
118 float curPeriod = _durationPeriod;
119 switch (_durationType)
120 {
121 case DurationType.Hours: return curPeriod * HoursRatio.Numerator;
122 case DurationType.Minutes: return curPeriod * MinutesRatio.Numerator;
123 case DurationType.Seconds: return curPeriod * SecondsRatio.Numerator;
124 case DurationType.Milliseconds: return curPeriod / MillisecondsRatio.Denominator;
125 case DurationType.Microseconds: return curPeriod / MicrosecondsRatio.Denominator;
126 case DurationType.Nanoseconds: return curPeriod / NanosecondsRatio.Denominator;
127 }
128
129 throw new InvalidOperationException("Cannot return a period unit that hasn't been changed.");
130 }
