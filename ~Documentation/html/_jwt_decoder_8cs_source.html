229 public void Validate(JwtParts jwt, params byte[][] keys)
230 {
231 if (jwt is null)
232 throw new ArgumentNullException(nameof(jwt));
233 if (jwt.Payload is null)
234 throw new ArgumentNullException(nameof(JwtParts.Payload));
235 if (jwt.Signature is null)
236 throw new ArgumentNullException(nameof(JwtParts.Signature));
237
238 var decodedPayload = GetString(_urlEncoder.Decode(jwt.Payload));
239 var decodedSignature = _urlEncoder.Decode(jwt.Signature);
240
241 var header = DecodeHeader<JwtHeader>(jwt);
242 var algorithm = _algFactory.Create(JwtDecoderContext.Create(header, decodedPayload, jwt));
243 if (algorithm is null)
244 throw new ArgumentNullException(nameof(algorithm));
245
246 var bytesToSign = GetBytes(jwt.Header, '.', jwt.Payload);
247
248 if (algorithm is IAsymmetricAlgorithm asymmAlg)
249 {
250 _jwtValidator.Validate(decodedPayload, asymmAlg, bytesToSign, decodedSignature);
251 }
252 else
253 {
254 ValidSymmetricAlgorithm(keys, decodedPayload, algorithm, bytesToSign, decodedSignature);
255 }
256 }

267 private void ValidSymmetricAlgorithm(byte[][] keys, string decodedPayload, IJwtAlgorithm algorithm, byte[] bytesToSign, byte[] decodedSignature)
268 {
269 if (keys is null)
270 throw new ArgumentNullException(nameof(keys));
271 if (!AllKeysHaveValues(keys))
272 throw new ArgumentOutOfRangeException(nameof(keys));
273
274 // the signature on the token, with the leading =
275 var rawSignature = Convert.ToBase64String(decodedSignature);
276
277 // the signatures re-created by the algorithm, with the leading =
278 var recreatedSignatures = keys.Select(key => Convert.ToBase64String(algorithm.Sign(key, bytesToSign))).ToArray();
279
280 _jwtValidator.Validate(decodedPayload, rawSignature, recreatedSignatures);
281 }
282
283 private static bool AllKeysHaveValues(byte[][] keys)
284 {
285 if (keys is null)
286 return false;
287
288 if (keys.Length == 0)
289 return false;
290
291 return Array.TrueForAll(keys, key => key.Length > 0);
292 }

294 private void ValidateNoneAlgorithm(JwtParts jwt)
295 {
296 var header = DecodeHeader<JwtHeader>(jwt);
297
298 if (String.IsNullOrEmpty(header.Type) &&
299 String.IsNullOrEmpty(header.Algorithm))
300 {
301 throw new InvalidOperationException("Error deserializing JWT header, all mandatory properties are null or empty");
302 }
303
304 if (String.Equals(header.Algorithm, nameof(JwtAlgorithmName.None), StringComparison.OrdinalIgnoreCase) &&
305 !String.IsNullOrEmpty(jwt.Signature))
306 {
307 throw new InvalidOperationException("Signature is not acceptable for algorithm None");
308 }
