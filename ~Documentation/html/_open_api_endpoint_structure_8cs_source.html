9public class OpenApiEndpointStructure : EndpointStructure<OpenApiUrl>
10{
11 public OpenApiEndpointStructure()
12 {
13 _allocatedId = Random.Range(900000, 10000000);
14 MgcxmObjectManager.Register(_allocatedId, this);
15 }
16 ~OpenApiEndpointStructure() => MgcxmObjectManager.Deregister(_allocatedId);
17
18 public override MgcxmHttpEndpoint ResolveUrl(string url, MgcxmHttpRequest request)
19 {
20 if (!url.StartsWith(ApiPath))
21 throw new OpenApiEndpointSearchException($"Cannot find '{url}'. The url does not start with '{ApiPath}' and is not a valid OpenAPI endpoint.");
22
23 foreach (var vEndpoint in Endpoints)
24 {
25 // resolve the endpoint
26 Logger.Trace($"Endpoint = {vEndpoint.Key.ToString()}, Url = {url}");
27 bool isMatched = MgcxmHttpEndpoint.Matches(vEndpoint.Key.ToString(), vEndpoint.Value, request);
28 if (isMatched) return vEndpoint.Value;
29 }
30 return null!;
31 }
32
33 public override void AddEndpoint(EndpointUrl url, HttpMethods method, OnEndpointRequested requestedHandler, MgcxmHttpListener listener)
34 {
35 var openApiUrl = (OpenApiUrl)url;
36 _allEndpoints.Add(openApiUrl, new MgcxmHttpEndpoint(listener.AllocatedId, openApiUrl.ToString(), new MgcxmDynamicIdentifier[0], method, requestedHandler));
37 }
38
39 public MgcxmId AllocatedId => _allocatedId;
40 public override string ApiPath => "/api/";
41 public override IReadOnlyDictionary<OpenApiUrl, MgcxmHttpEndpoint> Endpoints => _allEndpoints;
42
43 private MgcxmId _allocatedId;
44 private Dictionary<OpenApiUrl, MgcxmHttpEndpoint> _allEndpoints = new();
45}
