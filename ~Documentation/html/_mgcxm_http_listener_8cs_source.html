39 public MgcxmHttpListener(IpAddress addressToHostOn)
40 {
41 // Initialize the HttpListener with the provided IP address and port.
42 _listener = new HttpListener();
43 _listener.Prefixes.Add(
44 $"http://{AddressUtilities.FormatIpAddress(addressToHostOn.Origin)}:{addressToHostOn.Port}/");
45
46 // Generate an allocated ID for the listener.
47 _allocatedId = GetHashCode();
48
49 _endpoints = new List<MgcxmHttpEndpoint>();
50
51 // Create core endpoints, e.g., /favicon.ico.
52 InitializeCore();
53
54 // Register the listener object.
55 MgcxmObjectManager.Register(_allocatedId, this);
56
57 // Resolve and add all endpoints with the HttpEndpointAttribute.
58 AttributeResolver.ResolveAllMethod<HttpEndpointAttribute>(
59 GetType(),
60 typeof(HttpEndpointAttribute),
61 (method, attr) =>
62 {
63 // Check if an endpoint with the same URL already exists.
64 if (_endpoints.Any(x => x.Url == attr.Url))
65 return; // Cannot add endpoints with the same URL.
66
67 var dynamicUseAttribute = method.GetCustomAttribute<HttpDynamicUseAttribute>();
68
69 // Split the URL and find dynamic identifiers, if any.
70 string[] urlParts = attr.Url.Split("/");
71 List<MgcxmDynamicIdentifier> currentIdentifiers = new List<MgcxmDynamicIdentifier>();
72 for (int i = 0; i < urlParts.Length; i++)
73 {
74 if (dynamicUseAttribute != null && dynamicUseAttribute.DynamicIdentifiers.Contains(urlParts[i]))
75 {
76 currentIdentifiers.Add(
77 new MgcxmDynamicIdentifier(
78 i,
79 dynamicUseAttribute.DynamicIdentifiers.First(x => x == urlParts[i])));
80 }
81 }
82
83 // Set the return type for the attribute.
84 attr.ReturnType = method.ReturnType;
85
86 // Create a new endpoint and add it to the list.
87 _endpoints.Add(new MgcxmHttpEndpoint(
88 _allocatedId, attr.Url, currentIdentifiers.ToArray(), attr.HttpMethod,
89 (req, res) =>
90 {
91 var parameters = method.GetParameters();
92 bool isInvalidSig = parameters.Count() != 2 ||
93 (parameters[0].ParameterType != typeof(MgcxmHttpRequest) &&
94 parameters[1].ParameterType != typeof(MgcxmHttpResponse));
95
96 if (isInvalidSig) return; // Cannot add endpoints with invalid signatures
97 method.Invoke(this, new object[] { req, res });
98 }));
99
100 Logger.Trace(string.Format("Added '0x{0:x8}' to '0x{1:x8}'", attr.Url.GetHashCode(),
101 _allocatedId.Id));
102 });
103 }
104
108 ~MgcxmHttpListener() => MgcxmObjectManager.Deregister(_allocatedId);

127 private async Task StartListening()
128 {
129 this.PrintDebugInfo();
130
131 _listener.Start();
132 _listenToRequests = true;
133
134 while (_listenToRequests)
135 {
136 var httpContext = await _listener.GetContextAsync();
137
138 var httpRequest = httpContext.Request;
139 var httpResponse = httpContext.Response;
140
141 // build needed data
142 string query = "";
143 string url = httpRequest.Url!.PathAndQuery;
144
145 Uri baseAddress = new Uri($"{httpRequest.Url.Scheme}://{httpRequest.Url.Host}:{httpRequest.Url.Port}");
146 HttpMethods method = HttpMethods.UNKNOWN;
147 Dictionary<string, string> requestHeaders = new Dictionary<string, string>();
148 Dictionary<string, string> queryData = new Dictionary<string, string>();
149 Dictionary<string, string> formData = new Dictionary<string, string>();
150 List<byte> bodyData = new List<byte>();
151
152 Dictionary<MgcxmString, MgcxmString> responseHeaders = new Dictionary<MgcxmString, MgcxmString>();
153
154 // build can parse flag and method
155 bool canParseMethod = Enum.TryParse(typeof(HttpMethods), httpRequest.HttpMethod, out object? methodObj);
156 if (canParseMethod && methodObj != null)
157 method = (HttpMethods)methodObj;
158
159 // build headers
160 for (int i = 0; i < httpRequest.Headers.Count; i++)
161 requestHeaders.Add(
162 httpRequest.Headers.GetKey(i) ?? "Unknown",
163 httpRequest.Headers.GetValues(i)?[0] ?? "Unknown");
164
165 // build query
166 if (url.Contains("?"))
167 {
168 var urlParts = url.Split("?");
169
170 url = WebUtility.UrlDecode(urlParts[0]);
171 query = urlParts[1] ?? "";
172
173 foreach (var queryPart in query.Split("&"))
174 {
175 var kvp = queryPart.Split("=");
176 if (kvp.Length > 1)
177 queryData.Add(kvp[0], kvp[1]);
178 }
179 }
180
181 // build content type
182 var contentType = HttpContentTypeHelper.ResolveValue(httpRequest.ContentType!);
183
184 // build body
185 int ibyte;
186 while ((ibyte = httpRequest.InputStream.ReadByte()) != -1)
187 bodyData.Add((byte)ibyte);
188
189 if (contentType == HttpContentTypes.UrlencodedForm)
190 {
191 foreach (var formPart in Encoding.UTF8.GetString(bodyData.ToArray()).Split("&"))
192 {
193 var kvp = formPart.Split("=");
194 if (kvp.Length > 1)
195 formData.Add(kvp[0], kvp[1]);
196 }
197 }
198
199 // format url
200 if (url.EndsWith("/"))
201 url = url.Remove(url.Length - 1, 1);
202
203 // construct data
204 var requestData = MgcxmHttpRequest.New(
205 method,
206 baseAddress,
207 url,
208 contentType,
209 bodyData.ToArray(),
210 formData,
211 requestHeaders,
212 queryData,
213 _allocatedId);
214
215 var responseData = MgcxmHttpResponse.New(
216 (HttpStatusCodes)httpResponse.StatusCode,
217 "",
218 responseHeaders,
219 Array.Empty<byte>(),
220 _allocatedId);
221
222 //Logger.Info($"=========== Ws Request ===========");
223 //Logger.Info($"==================================");
224
225 // log request
226 Logger.Info($"||==----------- Http Request -----------==||");
227 Logger.Info($"Server Id = 0x{_allocatedId.Id:x8}");
228 Logger.Info($"Requested Url = {requestData.Uri}");
229 Logger.Info($"Http Method = {requestData.HttpMethod}");
230 if (requestData.HttpMethod == HttpMethods.POST || requestData.HttpMethod == HttpMethods.PUT)
231 {
232 Logger.Info($"Content Type = {requestData.ContentType}");
233 Logger.Info($"Content Length = {requestData.RawBodyData.Length}");
234 Logger.Info($"Content = {requestData.Body.Truncate(50, "...")}");
235 }
236
237 // add required headers
238 responseData.Header("X-Powered-By", "Mgcxm.Net");
239 responseData.Header("Vary", "Accept-Encoding");
240 responseData.Header("Request-Context", $"appId=cid-v1:{Guid.NewGuid().ToString()}");
241 responseData.Header("Pragma", "no-cache");
242 responseData.Header("Cache-Control", "no-cache");
243 responseData.Header("Expires", "-1");
244
245 httpResponse.KeepAlive = true;
246 responseData.Header("Connection", "keep-alive");
247
248 // write to response
249 if (_framework == null)
250 {
251 var endpoint = _endpoints.Find(x =>
252 x.Url == requestData.Uri && x.RequiredMethod == Enum.Parse<HttpMethods>(httpRequest.HttpMethod));
253 if (endpoint != null)
254 {
255 endpoint.OnEndpointRequested(requestData, responseData);
256 await responseData.Transfer(httpResponse);
257 }
258 else
259 {
260 // try using the Matches method
261 foreach (var vEndpoint in _endpoints)
262 {
263 try
264 {
265 if (MgcxmHttpEndpoint.Matches(requestData.Uri, vEndpoint, requestData))
266 {
267 // Logger.Trace($"0x{vEndpoint.EndpointId.Id:x8} meets dynamic criterion");
268 vEndpoint.OnEndpointRequested(requestData, responseData);
269 await responseData.Transfer(httpResponse);
270 }
271 else
272 {
273 // Logger.Trace($"0x{vEndpoint.EndpointId.Id:x8} does not meet dynamic criterion");
274 }
275 }
276 catch (Exception exception)
277 {
278 // Logger.Trace($"0x{vEndpoint.EndpointId.Id:x8} does not meet dynamic criterion");
279 Logger.Exception("Cannot parse Dynamic Url", exception);
280 }
281 }
282 }
283 }
284 else
285 {
286 try
287 {
288 await _framework.ResolveRequest(httpRequest, httpResponse, requestData, responseData);
289 }
290 catch (Exception exception)
291 {
292 Logger.Exception("Cannot resolve Framework endpoint", exception);
293 }
294 }
295 }
296
297 _listener.Stop();
298 }
