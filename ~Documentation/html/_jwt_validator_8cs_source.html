148 private Exception GetValidationException(string payloadJson)
149 {
150 if (String.IsNullOrEmpty(payloadJson))
151 throw new ArgumentException(nameof(payloadJson));
152
153 var payloadData = _jsonSerializer.Deserialize<Dictionary<string, object>>(payloadJson);
154
155 var now = _dateTimeProvider.GetNow();
156 var secondsSinceEpoch = UnixEpoch.GetSecondsSince(now);
157
158 Exception exception = null;
159
160 if (_valParams.ValidateExpirationTime)
161 {
162 exception = ValidateExpClaim(payloadData, secondsSinceEpoch);
163 }
164
165 if (_valParams.ValidateIssuedTime)
166 {
167 exception ??= ValidateNbfClaim(payloadData, secondsSinceEpoch);
168 }
169
170 return exception;
171 }

176 private static bool IsAnySignatureValid(string decodedCrypto, string[] decodedSignatures) =>

180 private static bool CompareCryptoWithSignature(string decodedCrypto, string decodedSignature)
181 {
182 if (decodedCrypto.Length != decodedSignature.Length)
183 return false;
184
185 var decodedCryptoBytes = GetBytes(decodedCrypto);
186 var decodedSignatureBytes = GetBytes(decodedSignature);
187
188 byte result = 0;
189 for (var i = 0; i < decodedCrypto.Length; i++)
190 {
191 result |= (byte)(decodedCryptoBytes[i] ^ decodedSignatureBytes[i]);
192 }
193
194 return result == 0;
195 }

203 private Exception ValidateExpClaim(IReadOnlyPayloadDictionary payloadData, double secondsSinceEpoch)
204 {
205 if (!payloadData.TryGetValue("exp", out var expObj))
206 return null;
207
208 if (expObj is null)
209 return new SignatureVerificationException("Claim 'exp' must be a number.");
210
211 double expValue;
212 try
213 {
214 expValue = Convert.ToDouble(expObj);
215 }
216 catch
217 {
218 return new SignatureVerificationException("Claim 'exp' must be a number.");
219 }
220
221 if (secondsSinceEpoch - _valParams.TimeMargin >= expValue)
222 {
223 return new TokenExpiredException("Token has expired.")
224 {
225 Expiration = UnixEpoch.Value.AddSeconds(expValue),
226 PayloadData = payloadData
227 };
228 }
229
230 return null;
231 }
232
238 private Exception ValidateNbfClaim(IReadOnlyPayloadDictionary payloadData, double secondsSinceEpoch)
239 {
240 if (!payloadData.TryGetValue("nbf", out var nbfObj))
241 return null;
242
243 if (nbfObj is null)
244 return new SignatureVerificationException("Claim 'nbf' must be a number.");
245
246 double nbfValue;
247 try
248 {
249 nbfValue = Convert.ToDouble(nbfObj);
250 }
251 catch
252 {
253 return new SignatureVerificationException("Claim 'nbf' must be a number.");
254 }
255
256 if (secondsSinceEpoch + _valParams.TimeMargin < nbfValue)
257 {
258 return new TokenNotYetValidException("Token is not yet valid.")
259 {
260 NotBefore = UnixEpoch.Value.AddSeconds(nbfValue),
261 PayloadData = payloadData
262 };
263 }
264
265 return null;
266 }
