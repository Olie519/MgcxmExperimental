52 public static bool Matches(string url, MgcxmHttpEndpoint endpoint, MgcxmHttpRequest request)
53 {
54 if (endpoint.RequiredMethod != request.HttpMethod)
55 return false;
56
57 // Check if the URL matches
58 string[] urlSegments = url.Split('/');
59 string[] endpointSegments = endpoint.Url.Split('/');
60
61 if (urlSegments.Length != endpointSegments.Length)
62 return false;
63
64 for (int i = 0; i < urlSegments.Length; i++)
65 {
66 string urlSegment = urlSegments[i];
67 string endpointSegment = endpointSegments[i];
68
69 // Check if the segments match exactly or if it's a dynamic identifier
70 if (urlSegment != endpointSegment && !IsDynamicIdentifier(endpointSegment))
71 return false;
72 }
73
74 return true;
75 }

82 private static bool IsDynamicIdentifier(string segment) => segment.StartsWith("{") && segment.EndsWith("}");
