111 public static string ResolveValue(HttpContentTypes contentType)
112 {
113 try
114 {
115 var type = typeof(HttpContentTypes);
116 var values = type.GetMember(contentType.ToString());
117 var enumValueMemberInfo = values.FirstOrDefault(m => m.DeclaringType == type);
118
119 CommonMimeTypeAttribute mimeTypeAttribute;
120 if ((mimeTypeAttribute = enumValueMemberInfo.GetCustomAttribute<CommonMimeTypeAttribute>()) == null)
121 throw new InvalidDataException("A common mime type is not attached to this enum.");
122
123 return mimeTypeAttribute.MimeType;
124 }
125 catch
126 {
127 return ResolveValue(HttpContentTypes.Invalid);
128 }
129 }

136 public static HttpContentTypes ResolveValue(string contentType)
137 {
138 var type = typeof(HttpContentTypes);
139 var values = type.GetMembers();
140
141 foreach (var value in values)
142 {
143 CommonMimeTypeAttribute mimeTypeAttribute = value.GetCustomAttribute<CommonMimeTypeAttribute>();
144 if (mimeTypeAttribute != null && mimeTypeAttribute.MimeType == contentType)
145 return Enum.Parse<HttpContentTypes>(value.Name);
146 }
147
148 return HttpContentTypes.Invalid;
149 }
