340 private void TryCreateEncoder()
341 {
342 if (_algorithm is null && _algFactory is null)
343 throw new InvalidOperationException($"Can't instantiate {nameof(JwtEncoder)}. Call {nameof(WithAlgorithm)}.");
344
345 var jsonSerializer = _jsonSerializerFactory.Create();
346 if (jsonSerializer is null)
347 throw new InvalidOperationException($"Can't instantiate {nameof(JwtEncoder)}. Call {nameof(WithJsonSerializer)}");
348 if (_urlEncoder is null)
349 throw new InvalidOperationException($"Can't instantiate {nameof(JwtEncoder)}. Call {nameof(WithUrlEncoder)}.");
350
351 if (_algorithm is object)
352 _encoder = new JwtEncoder(_algorithm, jsonSerializer, _urlEncoder);
353 else if (_algFactory is object)
354 _encoder = new JwtEncoder(_algFactory, jsonSerializer, _urlEncoder);
355 }
356
357 private void TryCreateDecoder()
358 {
359 TryCreateValidator();
360
361 var jsonSerializer = _jsonSerializerFactory.Create();
362 if (jsonSerializer is null)
363 throw new InvalidOperationException($"Can't instantiate {nameof(JwtDecoder)}. Call {nameof(WithJsonSerializer)}.");
364 if (_urlEncoder is null)
365 throw new InvalidOperationException($"Can't instantiate {nameof(JwtDecoder)}. Call {nameof(WithUrlEncoder)}.");
366
367 if (_algorithm is object)
368 _decoder = new JwtDecoder(jsonSerializer, _validator, _urlEncoder, _algorithm);
369 else if (_algFactory is object)
370 _decoder = new JwtDecoder(jsonSerializer, _validator, _urlEncoder, _algFactory);
371 else if (!_valParams.ValidateSignature)
372 _decoder = new JwtDecoder(jsonSerializer, _urlEncoder);
373 }

478 private string GetPropName(MemberInfo prop)
479 {
480 var jsonSerializer = _jsonSerializerFactory.Create();
481
482 var customAttributes = prop.GetCustomAttributes(inherit: true);
483 foreach (var attribute in customAttributes)
484 {
485 switch (jsonSerializer)
486 {
487 case JsonNetSerializer:
488 {
489 if (attribute is JsonPropertyAttribute jsonNetProperty)
490 {
491 return jsonNetProperty.PropertyName;
492 }
493 break;
494 }
495#if MODERN_DOTNET
496 case SystemTextSerializer:
497 {
498 if (attribute is JsonPropertyNameAttribute stjProperty)
499 {
500 return stjProperty.Name;
501 }
502 break;
503 }
504#endif
505 default:
506 {
507 return prop.Name;
508 }
509 }
510 }
511
512 return prop.Name;
513 }
