92 public async Task Transfer(HttpListenerResponse response)
93 {
94 response.StatusCode = (int)_statusCode;
95 foreach (var headerKvp in _headers)
96 response.Headers.Set(headerKvp.Key, headerKvp.Value);
97
98 response.ContentType = _contentType;
99 response.ContentLength64 = _responseData.Length;
100
101 var stream = response.OutputStream;
102 await stream.WriteAsync(_responseData, 0, _responseData.Length);
103 stream.Close();
104 }
